Solution - function Numberofvowels(str) {
		arr = str.split();
		var count = 0;
		for(i=0;i<=str.length;i++){
			if(arr[i] == 'a' || arr[i] == 'e' ||arr[i] == 'i' ||arr[i] == 'o' ||arr[i] == 'u') {
				count += 1;
			}}
		return count;
		}
