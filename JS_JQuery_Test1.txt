Question- 1
Solution - function Sumofarrays(arr_1,arr_2) {
		var arr1sum = 0;
		var arr2sum = 0;
		for(i=0;i<=arr1sum.length - 1;i++) {
			arr1sum += arr_1[i];		
		}
		for(i=0;i<=arr2sum.length - 1;i++) {
			arr2sum += arr_2[i];
			}	
		result = arr1sum + arr2sum;
		return result;
	}

Question - 7
Solution - function factorial(n) {
		if(n==0 || n==1){
			return 1; }
		else{	
			let facto = 1;
			for(i=1;i<=n;i++){
				facto *= i; 
			return facto;
		}

		}

Question - 9
Solution - function palindrome(str1,str2){
		rev1 = str1.split("").reverse().join();
		rev2 = str2.split("").reverse().join();
		if(str1 == rev1){
			return "Yes"; }
		else if(str1 != rev1){
			return "No";
		}
		if(str2 == rev2){
			return "Yes"; }
		else if(str2 != rev2){
			return "No";
		}

		
		}
